JS 작성

스크립트 부분입니다. input박스와 목록 리스트를 설정합니다. cssName을 searchList에 있는 모든 이름을 가져오는걸로 설정한뒤 cssName에 사용자가 찾는 단어를 포함한다면 hide를 remove하고, 포함하지않는다면 hide를 add해 보이지 않게 설정하였습니다. 

const searchBox = document.querySelector(".search__box input");   //input 박스
const searchList = document.querySelectorAll(".search__list ul li");    //목록 리스트

const searchInfo = document.querySelector(".search__info .num");

//속성 갯수 설정하기


searchInfo.textContent = searchList.length;

searchBox.addEventListener("keyup", ()=>{
    const searchWord = searchBox.value;   //사용자가 입력한 키워드

    searchList.forEach(el => {
        const cssName = el.dataset.name;
        // console.log(cssName);

        if(cssName.includes(searchWord)){
            el.classList.remove("hide");
        } else {
            el.classList.add("hide");
        }
    });
})

JS 작성

이번 스크립트는 엄청 간단합니다. 재귀함수(무한 함수)를 사용할것인데, 메모리를 적게 차지하는 requestAnimationFrame함수를 이용해 스크롤 때 마다 감지를하며, 스크롤 탑 값이 적정한 위치에 왔을 때 show(클래스)를 추가하는 식을 만들었습니다. 

function scroll(){
    let scrollTop = window.pageYOffset || document.documentElement.scrollTop || window.screenTop;

    document.querySelectorAll(".content__item").forEach(item => {
        if(scrollTop > item.offsetTop - window.innerHeight / 2 ){
            item.classList.add("show");
        }
    })
    requestAnimationFrame(scroll);  /*1분에 60번 실행하는 무한 함수(재귀함수)지만 메모리 적게 차지한다. */
}
scroll();

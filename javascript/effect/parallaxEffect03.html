let nowScroll = true;
let lastScroll = 0;
  
function scrollProgress(){                     /* nowScroll은 스크롤이 움직일 때(= scrollProgress 함수가 실행 될 떄) true가 된다. */
    nowScroll = true;
  
    setInterval(() => {
        if(nowScroll){
            nowScroll = false;                 /*nowScroll 을 false로 하는 이유는 스크롤이 움직였을 때만 함수를 실행시키기 위함 */
            hasScroll();
        }
    }, 300);
  }
  
function hasScroll(){
    let scrollTop = window.pageYOffset || window.scrollY || document.documentElement.scrollTop;
  
    if(scrollTop < lastScroll){            /* 스크롤 탑 값이 lastScroll값보다 작을 때 즉 스크롤을 위로 올라갈 때 실행 */
        document.querySelector("#parallax__nav").classList.add("show");
    } else {
        document.querySelector("#parallax__nav").classList.remove("show");
    }
    lastScroll = scrollTop;
  }
  
window.addEventListener("scroll", scrollProgress);
